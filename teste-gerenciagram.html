# Definir somente o ms de cada voo
df['Ms Partida Real'] = pd.DatetimeIndex(pd.to_datetime(df['Partida Real'], format='%d/%m/%Y %H:%M')).month

# Dados de apoio para o grfico
qtd_voo = df.groupby('Ms Partida Real').size()

# Grfico
plot = qtd_voo.plot(kind='bar', color='purple', title='Total de Atrasos Partindo de Salvador em 2018')
plot.set_xlabel('Ms')
plot.set_ylabel('Quantidade de Atrasos')

import pandas as pd
import numpy as np
import glob

# Importao das planilhas
all_csv = glob.glob(r'./data/anac/2018/*.csv')

li = []

for csv in all_csv:
    df_temp = pd.read_csv(csv, encoding='ISO-8859-1', sep=';')
    li.append(df_temp)

df = pd.concat(li, sort=False)

# Definir os voos com atrasos na partida e/ou chegada
df['Atraso Partida'] = np.where(pd.to_datetime(df['Partida Real'], format='%d/%m/%Y %H:%M') - pd.to_datetime(df['Partida Prevista'], format='%d/%m/%Y %H:%M') > '00:00:00', 'SIM', 'NAO')
df['Atraso Chegada'] = np.where(pd.to_datetime(df['Chegada Real'], format='%d/%m/%Y %H:%M') - pd.to_datetime(df['Chegada Prevista'], format='%d/%m/%Y %H:%M') > '00:00:00', 'SIM', 'NAO')

# Filtragem dos dados
df = df[(df['Situao Voo'] == 'REALIZADO') & (df['ICAO Aerdromo Origem'] == 'SBSV') & (df['ICAO Aerdromo Destino'] == 'SBSP') & (df['Atraso Partida'] == 'NAO') & (df['Atraso Chegada'] == 'NAO')]

# Definir a hora de partida de cada voo
df['Hora Partida'] = pd.DatetimeIndex(pd.to_datetime(df['Partida Real'], format='%d/%m/%Y %H:%M')).hour

# Dados de apoio para o grfico
horario = df.groupby('Hora Partida').size()

# Grfico
plot = horario.plot(kind='line', marker='*', markersize='7', color='purple', title='Melhor Horrio para Viajar de Salvador para So Paulo')
plot.set_xlabel('Horrios')
plot.set_ylabel('Ocorrncias de Voo')

import pandas as pd
import numpy as np
import glob
import matplotlib.pyplot as plot

# Importao das planilhas
all_csv = glob.glob(r'./data/anac/outros/*.csv')

li = []

for csv in all_csv:
    df_temp = pd.read_csv(csv, encoding='ISO-8859-1', sep=';')
    li.append(df_temp)

df = pd.concat(li, sort=False)

# Definir quais voos tiveram atraso na chegada
df['Atraso Chegada'] = np.where(pd.to_datetime(df['Chegada Real'], format='%d/%m/%Y %H:%M') - pd.to_datetime(df['Chegada Prevista'], format='%d/%m/%Y %H:%M') > '00:00:00', 'SIM', 'NAO')
df['Atraso Partida'] = np.where(pd.to_datetime(df['Partida Real'], format='%d/%m/%Y %H:%M') - pd.to_datetime(df['Partida Prevista'], format='%d/%m/%Y %H:%M') > '00:00:00', 'SIM', 'NAO')

# Definir ano e ms de cada voo
df['Ano'] = pd.DatetimeIndex(pd.to_datetime(df['Partida Real'], format='%d/%m/%Y %H:%M')).year
df['Ms'] = pd.DatetimeIndex(pd.to_datetime(df['Partida Real'], format='%d/%m/%Y %H:%M')).month

# Filtragem e preparao dos dados
df = df[(df['Situao Voo'] == 'REALIZADO') & (df['Ano'] > 2015) & (df['Ms'] >= 1) & (df['Ms'] <= 2)]
df['Ano'] = df['Ano'].astype(int)

## Atraso na Partida
# Dados de apoio para o grfico de partida
qtd_partida = df.loc[df['Atraso Partida'] == 'SIM'].groupby(['Ano','Ms']).size().unstack()

plot_partida.set_xlabel('Ano')
